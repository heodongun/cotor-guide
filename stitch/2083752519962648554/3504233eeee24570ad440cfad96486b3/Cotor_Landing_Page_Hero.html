Get Started arrow_forward description View Documentation
terminal npm install -g cotor-cli
cotor.yaml
version: '2.0'
system:
  name: "Research-Cluster"
  strategy: "hierarchical"
  master:
    model: "gpt-4-turbo"
    capabilities: [delegation, synthesis]
agents:
  - name: "Data-Gatherer"
    role: "Web Scraping & API Interaction"
    tools: [browser, serp_api]
    concurrency: 5
  - name: "Analyst"
    role: "Data Processing & Summarization"
    dependencies: ["Data-Gatherer"]
# Run with: cotor up -f cotor.yaml
